  METHOD HANDLE_DATA_CHANGED.
    DATA:MOD_DATA    TYPE LVC_T_MODI,
         WA_MOD_DATA TYPE LVC_S_MODI,
         IT_ITEMC    LIKE TABLE OF IT_ITEM.
    DATA:FLAG TYPE CHAR1.
    CLEAR:MOD_DATA,WA_MOD_DATA.
    MOD_DATA = ER_DATA_CHANGED->MT_MOD_CELLS.
    LOOP AT MOD_DATA INTO WA_MOD_DATA.
      READ TABLE IT_ITEM INTO WA_ITEM INDEX WA_MOD_DATA-ROW_ID.
      CHECK SY-SUBRC = 0.
      CASE WA_MOD_DATA-FIELDNAME.
        WHEN 'ZDHL'.
          ASSIGN COMPONENT WA_MOD_DATA-FIELDNAME OF STRUCTURE WA_ITEM TO <FS>.
          IF SY-SUBRC EQ 0.
            PERFORM CHECKMENGE CHANGING WA_MOD_DATA-VALUE FLAG.
            IF FLAG = 'E'.
              MESSAGE S057 WITH WA_MOD_DATA-VALUE DISPLAY LIKE 'E'.
              EXIT.
            ENDIF.
            PERFORM DELQFW CHANGING WA_MOD_DATA-VALUE.
            <FS> = WA_MOD_DATA-VALUE.
            MODIFY IT_ITEM FROM WA_ITEM  INDEX WA_MOD_DATA-ROW_ID TRANSPORTING ZDHL.
          ENDIF.
          PERFORM CALZZL USING 'ZDHL'.
          CALL METHOD CL_GUI_CFW=>SET_NEW_OK_CODE
            EXPORTING
              NEW_CODE = 'ITEMCHG'.
        WHEN 'ZBZRQ' OR 'ZSCRQ' OR 'ZRKL' OR 'ZJSL'.
          ASSIGN COMPONENT WA_MOD_DATA-FIELDNAME OF STRUCTURE WA_ITEM TO <FS>.
          IF SY-SUBRC EQ 0.
            CASE WA_MOD_DATA-FIELDNAME.
              WHEN 'ZRKL' OR 'ZJSL'.
                PERFORM CHECKMENGE CHANGING WA_MOD_DATA-VALUE FLAG.
                IF FLAG = 'E'.
                  MESSAGE S057 WITH WA_MOD_DATA-VALUE DISPLAY LIKE 'E'.
                  EXIT.
                ENDIF.
                PERFORM DELQFW CHANGING WA_MOD_DATA-VALUE.
                IF WA_MOD_DATA-FIELDNAME = 'ZJSL'.
                  IF WA_ITEM-MEINS = WA_ITEM-RKDW.
                    PERFORM DELQFW CHANGING WA_MOD_DATA-VALUE.
                    WA_ITEM-ZRKL = WA_MOD_DATA-VALUE.
                  ENDIF.
                ENDIF.
              WHEN 'ZSCRQ'.
                WA_MOD_DATA-VALUE = WA_MOD_DATA-VALUE+0(4)
                                 && WA_MOD_DATA-VALUE+5(2)
                                 && WA_MOD_DATA-VALUE+8(2).
            ENDCASE.
            <FS> = WA_MOD_DATA-VALUE.
            MODIFY IT_ITEM FROM WA_ITEM  INDEX WA_MOD_DATA-ROW_ID.
          ENDIF.
          PERFORM CALZTMM202 TABLES IT_ITEM USING WA_MOD_DATA-FIELDNAME.
        WHEN 'LGORT'.
          CLEAR WA_ITEM-LGOBE.
          READ TABLE IT_T001L INTO T001L WITH KEY WERKS = WA_ITEM-WERKS
                                                  LGORT = WA_MOD_DATA-VALUE
                                                  BINARY SEARCH.
          IF SY-SUBRC EQ 0.
            WA_ITEM-LGOBE = T001L-LGOBE.
          ENDIF.
          MODIFY IT_ITEM FROM WA_ITEM INDEX WA_MOD_DATA-ROW_ID TRANSPORTING LGOBE.
        WHEN 'ZXHPC'.
          CLEAR:ZVMMPO,EKKN.
          SELECT SINGLE *
            INTO ZVMMPO
            FROM ZVMMPO
            WHERE EBELN = WA_ITEM-EBELN
            AND   EBELP = WA_ITEM-EBELP.
          SELECT SINGLE *
            INTO EKKN
            FROM EKKN
            WHERE EBELN = ZVMMPO-EBELN
            AND   EBELP = ZVMMPO-EBELP.
          IF SY-SUBRC EQ 0.
            IF ZVMMPO-KNTTP = 'Q'.
              SELECT SINGLE RDLAB
                INTO @WA_ITEM-ZXHL
                FROM MSRD
                WHERE WERKS = @WA_ITEM-WERKS
                AND   MATNR = @WA_ITEM-MATNR
                AND   LIFNR = @WA_HEAD-LIFNR
                AND   CHARG = @WA_MOD_DATA-VALUE
                AND   PSPNR = @EKKN-PS_PSP_PNR
                AND   RDLAB GT 0.
            ELSE.
              SELECT SINGLE LBLAB
                INTO @WA_ITEM-ZXHL
                FROM MSLB
                WHERE WERKS = @WA_ITEM-WERKS
                AND   MATNR = @WA_ITEM-MATNR
                AND   LIFNR = @WA_HEAD-LIFNR
                AND   CHARG = @WA_MOD_DATA-VALUE
                AND   LBLAB GT 0.
            ENDIF.
            MODIFY IT_ITEM FROM WA_ITEM INDEX WA_MOD_DATA-ROW_ID TRANSPORTING ZXHL.
          ENDIF.
      ENDCASE.
    ENDLOOP.
    PERFORM REFRESHALV.
  ENDMETHOD.                    "handle_data_changed

  METHOD HANDLE_HOTSPOT_CLICK.
    DATA:WA LIKE LINE OF IT_ITEM.
    FIELD-SYMBOLS:<FSE> TYPE ANY,
                  <WAE> TYPE ANY.
    CONDENSE E_ROW_ID     NO-GAPS.
    CONDENSE E_COLUMN_ID  NO-GAPS.
    READ TABLE IT_ITEM ASSIGNING <WAE> INDEX E_ROW_ID.
    ASSIGN COMPONENT E_COLUMN_ID OF STRUCTURE <WAE> TO <FSE>.
    CHECK SY-SUBRC = 0.
    IF <FSE> = 'X'.
      <FSE> = ''.
    ELSE.
      <FSE> = 'X'.
    ENDIF.
    PERFORM REFRESHALV.
  ENDMETHOD.

  METHOD HANDLE_ON_F4.
    ER_EVENT_DATA->M_EVENT_HANDLED = 'X'.
    ASSIGN ER_EVENT_DATA->M_DATA->* TO <IT_MODI>.
    READ TABLE IT_ITEM INTO WA_ITEM INDEX ES_ROW_NO-ROW_ID.
    CHECK SY-SUBRC = 0.
    CASE E_FIELDNAME.
      WHEN 'ZXHPC' OR 'RKDW' OR 'LGORT' OR 'ZJDW'.
        PERFORM F4HELP TABLES RETURN_TAB USING  E_FIELDNAME 'IT_ITEM' WA_ITEM.
    ENDCASE.
    IF RETURN_TAB[] IS NOT INITIAL.
      READ TABLE RETURN_TAB INTO WA_RETURN_TAB INDEX 1.
      IF SY-SUBRC EQ 0.
        WA_MODI-ROW_ID = ES_ROW_NO-ROW_ID.
        WA_MODI-FIELDNAME = E_FIELDNAME.
        WA_MODI-VALUE = WA_RETURN_TAB-FIELDVAL.
        APPEND WA_MODI TO <IT_MODI>.
      ENDIF.
    ENDIF.
    PERFORM REFRESHALV.
  ENDMETHOD.
