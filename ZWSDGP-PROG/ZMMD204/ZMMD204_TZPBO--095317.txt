*&---------------------------------------------------------------------*
*& Module MDL_CHANGE_INPUT OUTPUT
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
MODULE mdl_change_input OUTPUT.

  LOOP AT SCREEN .
    IF it_tz-zjsfs = 'DJ' AND screen-name = 'IT_TZ-ZWLDJ'.
      screen-input = 1 .
    ELSEIF  it_tz-zjsfs = 'ZC' AND screen-name = 'IT_TZ-ZYSJE'.
      screen-input = 1 .
    ENDIF.
    MODIFY SCREEN.
  ENDLOOP.

ENDMODULE.
