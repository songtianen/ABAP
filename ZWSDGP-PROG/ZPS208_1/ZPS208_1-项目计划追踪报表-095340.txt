FORM getdata.
  CLEAR: item, item[], zboitem, zboitem[].
  SELECT
    *
    INTO CORRESPONDING FIELDS OF TABLE item
    FROM ztps013a
  WHERE pspid = p_pspid.

  IF item[] IS NOT INITIAL.
    LOOP AT item.
      SELECT
      SINGLE
      pspid,
      zdtbh,
      zrwbh,
      zcpbh,
      MAX( zbbh ) AS zbbh
      FROM ztps013b
      INTO  CORRESPONDING FIELDS OF @zboitem
      WHERE pspid = @item-pspid
      AND zdtbh = @item-zdtbh
      GROUP BY pspid, zdtbh, zrwbh, zcpbh.
      APPEND zboitem.
      CLEAR: zboitem.
    ENDLOOP.
    LOOP AT zboitem.
      SELECT SINGLE
        *
        INTO CORRESPONDING FIELDS OF zboitem
        FROM ztps013b
        WHERE pspid = p_pspid
        AND zdtbh = zboitem-zdtbh
        AND zrwbh = zboitem-zrwbh
        AND zcpbh = zboitem-zcpbh
      AND zbbh = zboitem-zbbh.
      IF sy-subrc = 0.
        MODIFY zboitem.
      ENDIF.
    ENDLOOP.
  ENDIF.

ENDFORM.

FORM f4_p_pspid.
  SELECT
    ztps010a~kunnr,
    kna1~name1,
    ztps010a~pspid
    INTO TABLE @DATA(lt_kunnr)
    FROM ztps010a
    INNER JOIN kna1 ON kna1~kunnr = ztps010a~kunnr
  WHERE pspid = @p_pspid.
  CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
    EXPORTING
      retfield         = 'KUNNR'
      dynpprog         = sy-repid
      dynpnr           = sy-dynnr
      dynprofield      = 'P_KUNNR'
      value_org        = 'S'
      callback_program = sy-repid
      "callback_form    = 'CB_FORM'
      display          = 'F'   " <C Force
    TABLES
      value_tab        = lt_kunnr
      "return_tab       = return_tab
    EXCEPTIONS
      parameter_error  = 1
      no_values_found  = 2
      OTHERS           = 3.

ENDFORM.
