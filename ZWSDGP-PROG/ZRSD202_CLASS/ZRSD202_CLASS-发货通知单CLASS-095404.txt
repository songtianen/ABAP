  METHOD HANDLE_DATA_CHANGED.
    DATA:MOD_DATA    TYPE LVC_T_MODI,
         WA_MOD_DATA TYPE LVC_S_MODI.
    DATA:WA_ITEM LIKE LINE OF IT_ITEM.
    CLEAR:MOD_DATA,WA_MOD_DATA.
    MOD_DATA = ER_DATA_CHANGED->MT_MOD_CELLS.
    LOOP AT MOD_DATA INTO WA_MOD_DATA.
      CLEAR WA_ITEM.
      READ TABLE IT_ITEM INTO WA_ITEM INDEX WA_MOD_DATA-ROW_ID.
      CASE WA_MOD_DATA-FIELDNAME.
        WHEN 'ZZHWZT1'.
          IF WA_MOD_DATA-VALUE IS INITIAL.
            CLEAR:WA_ITEM-ZZHWZ,WA_ITEM-ZZHWZT.
          ELSE.
            SPLIT WA_MOD_DATA-VALUE AT '-' INTO WA_ITEM-ZZHWZ
            WA_ITEM-ZZHWZT.
          ENDIF.
        WHEN 'LGORT'.
          READ TABLE IT_T001L INTO IT_T001L WITH KEY WERKS = WA_ITEM-WERKS
                                            LGORT = WA_MOD_DATA-VALUE
                                            BINARY SEARCH.
          IF SY-SUBRC EQ 0.
            WA_ITEM-LGOBE = IT_T001L-LGOBE.
          ENDIF.
      ENDCASE.
      MODIFY IT_ITEM FROM WA_ITEM INDEX WA_MOD_DATA-ROW_ID.
    ENDLOOP.
    PERFORM REFRESHALV.
  ENDMETHOD.                    "handle_data_changed

  METHOD HANDLE_HOTSPOT_CLICK.
    DATA:WA LIKE LINE OF IT_ITEM.
    CONDENSE E_ROW_ID     NO-GAPS.
    CONDENSE E_COLUMN_ID  NO-GAPS.
    CLEAR WA.
    CASE E_COLUMN_ID.
      WHEN 'CHBOX'.
        READ TABLE IT_ITEM INTO WA INDEX E_ROW_ID.
        IF WA-CHBOX = 'X'.
          WA-CHBOX = ''.
        ELSE.
          WA-CHBOX = 'X'.
        ENDIF.
        MODIFY IT_ITEM FROM WA INDEX E_ROW_ID TRANSPORTING CHBOX.
      WHEN 'VGBEL'.
        READ TABLE IT_ITEM INTO WA INDEX E_ROW_ID.
        CHECK WA-VGBEL IS NOT INITIAL.
        CALL FUNCTION 'ZFM_CALLSODJ'
          EXPORTING
            VBELN  = WA-VGBEL
            INTYPE = 'HT'
            MODE   = 'S'.
      WHEN 'VBELV'.
        READ TABLE IT_ITEM INTO WA INDEX E_ROW_ID.
        CHECK WA-VBELV IS NOT INITIAL.
        CALL FUNCTION 'ZFM_CALLSODJ'
          EXPORTING
            VBELN  = WA-VBELV
            INTYPE = 'CK'
            MODE   = 'S'.
    ENDCASE.
    PERFORM REFRESHALV.
  ENDMETHOD.
