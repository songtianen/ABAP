FUNCTION zfm_gp_pp_oa_pcd_send.
*"----------------------------------------------------------------------
*"*"本地接口：
*"  IMPORTING
*"     VALUE(P_NAME) TYPE  STRING
*"  EXPORTING
*"     VALUE(P_OUTPUT) TYPE  STRING
*"     VALUE(P_STATUS) TYPE  I
*"     VALUE(P_RESULT) TYPE  STRING
*"     VALUE(RTYPE) TYPE  BAPI_MTYPE
*"     VALUE(RTMSG) TYPE  BAPI_MSG
*"  TABLES
*"      IN_TAB STRUCTURE  ZSPP_218
*"      T_RESULT STRUCTURE  ZSMM_017 OPTIONAL
*"----------------------------------------------------------------------
  zfmdatasave1 'ZFM_GP_PP_OA_PCD_SEND'.
  zfmdatasave2 'B'.
  COMMIT WORK.
  TYPES:BEGIN OF ty_out .
          INCLUDE TYPE zspp_218.
  TYPES:
        END OF ty_out.
  DATA:BEGIN OF wa_input1,
         intab TYPE TABLE OF ty_out,
       END OF wa_input1,
       BEGIN OF wa_input,
         data LIKE wa_input1,
       END OF wa_input,
       BEGIN OF wa_output1,
         tresult TYPE TABLE OF zsmm_017,
       END OF wa_output1,
       BEGIN OF wa_output,
         data LIKE wa_output1,
       END OF wa_output,
       wa_out LIKE LINE OF wa_input1-intab.


  CLEAR:oastr1,oastr2,oaurl,wa_input,wa_output,
  oahead[],oasta,oamsg.
*URL改为可配置，ZMM000
  PERFORM getdata(zpub_data) USING 'ZFM_GP_PP_OA_PCD_SEND' CHANGING oaurl.

  IF oaurl IS INITIAL.
    fillmsgr 'E' '请前往zmm000配置ZFM_GP_PP_OA_PCD_SEND的oaURL'.
  ENDIF.
  IF in_tab[] IS INITIAL.
    fillmsgr 'E' '入参表不能为空'.
  ENDIF.
  LOOP AT in_tab.
    CLEAR:wa_out.
    MOVE-CORRESPONDING in_tab TO wa_out.
    APPEND wa_out TO wa_input1-intab.
    MODIFY in_tab.
  ENDLOOP.

*根据传入数据转JSON
  wa_input-data-intab = wa_input1-intab.
*SAP结构转JSON
  CALL FUNCTION 'ZFMS_14_JSON'
    CHANGING
      jsonstr = oastr1
      data    = wa_input
    EXCEPTIONS
      OTHERS  = 1.

  SHIFT oastr1 LEFT  DELETING LEADING  '{'.

  CONCATENATE '{"senderLoginName":"'
*  CONCATENATE '{"SENDER":"'
  p_name
  '",'
  oastr1 INTO oastr1.

  p_result = oastr1.


*获取TOKEN
  CALL FUNCTION 'ZFM_GETOATOKEN'
    IMPORTING
      output = oastr2
    EXCEPTIONS
      OTHERS = 1.
  "zfmdatasave2 'R'.
  IF oastr2 IS INITIAL.
    fillmsgr 'E' '获取OA token失败'.
  ENDIF.
*填入HEADER
  CLEAR:oamsg,oasta.
  REFRESH:oahead.
  CLEAR oahead.
  oahead-name = 'token'.
  oahead-value = oastr2.
  APPEND oahead.
*调用函数HTTP
  CLEAR oastr2.

  PERFORM replace(zpubform) USING 'https' 'http' CHANGING oaurl.
  CALL FUNCTION 'ZFMS_15_HTTP'
    EXPORTING
      input     = oastr1
      url       = oaurl
      reqmethod = 'POST' "HTTP 方法
      http1_1   = 'X' "协议1.1/1.0
    IMPORTING
      output    = oastr2 "返回JSON报文
      rtmsg     = oamsg "消息
      status    = oasta "HTTP状态
    TABLES
      header    = oahead
    EXCEPTIONS
      OTHERS    = 1.
*返回报文转为SAP结构
  CALL FUNCTION 'ZFMS_14_JSON'
    CHANGING
      jsonstr = oastr2
      data    = wa_output
    EXCEPTIONS
      OTHERS  = 1.

  t_result[] = wa_output-data-tresult.
  p_status = oasta.
  p_output = oastr2.
  IF p_status = '200'.
    fillmsgr 'S' '推送oa成功'.
  ELSE.
    DATA(cc) = '推送oa失败:' && p_output.
    fillmsgr 'E' cc.
  ENDIF.
  zfmdatasave2 'R'.
ENDFUNCTION.
