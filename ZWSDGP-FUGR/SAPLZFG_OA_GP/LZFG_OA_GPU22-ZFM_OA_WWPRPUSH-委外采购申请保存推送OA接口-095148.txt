FUNCTION ZFM_OA_WWPRPUSH.
*"----------------------------------------------------------------------
*"*"本地接口：
*"  IMPORTING
*"     VALUE(P_NAME) TYPE  STRING
*"  EXPORTING
*"     VALUE(P_OUTPUT) TYPE  STRING
*"     VALUE(P_STATUS) TYPE  I
*"     VALUE(P_RESULT) TYPE  STRING
*"     VALUE(RTMSG) TYPE  BAPI_MSG
*"  TABLES
*"      IN_TAB STRUCTURE  ZSMM_208
*"      T_RESULT STRUCTURE  ZSMM_017
*"----------------------------------------------------------------------
  ZFMDATASAVE1 'ZFM_OA_WWPRPUSH'.
  ZFMDATASAVE2 'B'.
  COMMIT WORK.
  TYPES:BEGIN OF TY_OUT .
          INCLUDE TYPE ZSMM_208.
  TYPES:  ZBZ    TYPE STRING,
        END OF TY_OUT.
  DATA:BEGIN OF WA_INPUT1,
         INTAB TYPE TABLE OF TY_OUT,
       END OF WA_INPUT1,
       BEGIN OF WA_INPUT,
         DATA LIKE WA_INPUT1,
       END OF WA_INPUT,
       BEGIN OF WA_OUTPUT1,
         TRESULT TYPE TABLE OF ZSMM_017,
       END OF WA_OUTPUT1,
       BEGIN OF WA_OUTPUT,
         DATA LIKE WA_OUTPUT1,
       END OF WA_OUTPUT,
       WA_OUT LIKE LINE OF WA_INPUT1-INTAB.


  CLEAR:OASTR1,OASTR2,OAURL,WA_INPUT,WA_OUTPUT,
  OAHEAD[],OASTA,OAMSG.
*URL改为可配置，ZMM000
  PERFORM GETDATA(ZPUB_DATA) USING 'ZFM_OA_WWPRPUSH' CHANGING OAURL.

  CHECK OAURL IS NOT INITIAL
  AND IN_TAB[] IS NOT INITIAL.

  LOOP AT IN_TAB.
    CLEAR:WA_OUT.
    MOVE-CORRESPONDING IN_TAB TO WA_OUT.
    PERFORM GETLONGTEXT(ZPUBFORM) USING 'B01' IN_TAB-BANFN 'EBNAH'
          CHANGING WA_OUT-ZBZ.
    PERFORM GETLONGTEXT(ZPUBFORM) USING 'GRUN' IN_TAB-MATNR 'MATERIAL'
      CHANGING WA_OUT-MAKTX.
    PERFORM GETLONGTEXT(ZPUBFORM) USING 'GRUN' IN_TAB-MATNR1 'MATERIAL'
  CHANGING WA_OUT-MAKTX1.
    APPEND WA_OUT TO WA_INPUT1-INTAB.
    MODIFY IN_TAB.
  ENDLOOP.

*根据传入数据转JSON
  WA_INPUT-DATA-INTAB = WA_INPUT1-INTAB.
*SAP结构转JSON
  CALL FUNCTION 'ZFMS_14_JSON'
    CHANGING
      JSONSTR = OASTR1
      DATA    = WA_INPUT
    EXCEPTIONS
      OTHERS  = 1.

  SHIFT OASTR1 LEFT  DELETING LEADING  '{'.

  CONCATENATE '{"senderLoginName":"'
*  CONCATENATE '{"SENDER":"'
  P_NAME
  '",'
  OASTR1 INTO OASTR1.

  P_RESULT = OASTR1.


*获取TOKEN
  CALL FUNCTION 'ZFM_GETOATOKEN'
    IMPORTING
      OUTPUT = OASTR2
    EXCEPTIONS
      OTHERS = 1.
  "zfmdatasave2 'R'.
  CHECK OASTR2 IS NOT INITIAL.
*填入HEADER
  CLEAR:OAMSG,OASTA.
  REFRESH:OAHEAD.
  CLEAR OAHEAD.
  OAHEAD-NAME = 'token'.
  OAHEAD-VALUE = OASTR2.
  APPEND OAHEAD.
*调用函数HTTP
  CLEAR OASTR2.

  PERFORM REPLACE(ZPUBFORM) USING 'https' 'http' CHANGING OAURL.
  CALL FUNCTION 'ZFMS_15_HTTP'
    EXPORTING
      INPUT     = OASTR1
      URL       = OAURL
      REQMETHOD = 'POST' "HTTP 方法
      HTTP1_1   = 'X' "协议1.1/1.0
    IMPORTING
      OUTPUT    = OASTR2 "返回JSON报文
      RTMSG     = OAMSG "消息
      STATUS    = OASTA "HTTP状态
    TABLES
      HEADER    = OAHEAD
    EXCEPTIONS
      OTHERS    = 1.
*返回报文转为SAP结构
  CALL FUNCTION 'ZFMS_14_JSON'
    CHANGING
      JSONSTR = OASTR2
      DATA    = WA_OUTPUT
    EXCEPTIONS
      OTHERS  = 1.

  T_RESULT[] = WA_OUTPUT-DATA-TRESULT.
  P_STATUS = OASTA.
  P_OUTPUT = OAMSG.
  IF P_STATUS = '200'.
*自动审批
    CLEAR IN_TAB.
    READ TABLE IN_TAB INDEX 1.
    IF SY-SUBRC EQ 0.
      SELECT SINGLE *
        INTO @DATA(WA_EBAN)
        FROM EBAN
        WHERE BANFN = @IN_TAB-BANFN.
      CASE WA_EBAN-FRGKZ.
        WHEN '' OR 'B'.
          PERFORM RELPOPR(ZPUBFORM) USING WA_EBAN-BANFN
                                          'Z1'
                                          ''
                                          'PR'
                                    CHANGING RTMSG.
      ENDCASE.
    ENDIF.
  ENDIF.
*  P_RESULT = OASTR2.
  ZFMDATASAVE2 'R'.
ENDFUNCTION.
