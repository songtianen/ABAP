FORM T130F_LESEN_KOMPLETT.

  DESCRIBE TABLE IT130F LINES ZAEHLER.
  IF ZAEHLER EQ 0.
    CALL FUNCTION 'T130F_ARRAY_READ'
         TABLES
              TT130F = IT130F.
    LOOP AT IT130F.
*Tabelle der Felder mit SFGRUP aufbauen
      IF NOT IT130F-SFGRU  IS INITIAL.
          FTAB_SFGRUP-FNAME  = IT130F-FNAME.
          APPEND FTAB_SFGRUP.
      ENDIF.
    ENDLOOP.
  ENDIF.

ENDFORM.
