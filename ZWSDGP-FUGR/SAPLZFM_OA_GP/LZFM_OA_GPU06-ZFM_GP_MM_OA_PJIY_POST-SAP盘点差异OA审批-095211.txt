
FUNCTION ZFM_GP_MM_OA_PJIY_POST.
*"----------------------------------------------------------------------
*"*"本地接口：
*"  IMPORTING
*"     VALUE(P_NAME) TYPE  STRING OPTIONAL
*"  EXPORTING
*"     VALUE(P_OUTPUT) TYPE  STRING
*"     VALUE(P_STATUS) TYPE  I
*"     VALUE(P_RESULT) TYPE  STRING
*"  TABLES
*"      IN_TAB STRUCTURE  ZMM207
*"      T_RESULT STRUCTURE  ZSMM_017
*"----------------------------------------------------------------------
  ZFMDATASAVE1 'ZFM_GP_MM_OA_PJIY_POST'.
  ZFMDATASAVE2 'B'.

*  TYPES:BEGIN OF ty_out .
*          INCLUDE TYPE ztmm013a.
*  TYPES:  f01 TYPE string,
*          f02 TYPE string,
*        END OF ty_out.
  DATA:BEGIN OF WA_INPUT1,
         INTAB TYPE TABLE OF ZMM207,
       END OF WA_INPUT1,
       BEGIN OF WA_INPUT,
         DATA LIKE WA_INPUT1,
       END OF WA_INPUT,
       BEGIN OF WA_OUTPUT1,
         TRESULT TYPE TABLE OF ZSMM_017,
       END OF WA_OUTPUT1,
       BEGIN OF WA_OUTPUT,
         DATA LIKE WA_OUTPUT1,
       END OF WA_OUTPUT,
       WA_OUT LIKE LINE OF WA_INPUT1-INTAB.


  CLEAR:OASTR1,OASTR2,OAURL,WA_INPUT,WA_OUTPUT,
  OAHEAD[],OASTA,OAMSG.
*URL改为可配置，ZMM000
  PERFORM GETDATA(ZPUB_DATA) USING 'ZFM_GP_MM_OA_PJIY_POST' CHANGING OAURL.
  CHECK OAURL IS NOT INITIAL AND IN_TAB[] IS NOT INITIAL.
*根据传入数据转json
*  wa_input-data-intab = wa_input1-intab.
  LOOP AT IN_TAB.
    CONDENSE IN_TAB-ZPDUL NO-GAPS.
    CONDENSE IN_TAB-CLABS NO-GAPS.
    CONDENSE IN_TAB-ZBFSL NO-GAPS.
    CONDENSE IN_TAB-ZIYSL NO-GAPS.
    MODIFY IN_TAB.
  ENDLOOP.
  WA_INPUT-DATA-INTAB = IN_TAB[].
*SAP结构转JSON
  CALL FUNCTION 'ZFMS_14_JSON'
    CHANGING
      JSONSTR = OASTR1
      DATA    = WA_INPUT
    EXCEPTIONS
      OTHERS  = 1.

  SHIFT OASTR1 LEFT  DELETING LEADING  '{'.

  CONCATENATE '{"senderLoginName":"'
*  CONCATENATE '{"SENDER":"'
  P_NAME
  '",'
  OASTR1 INTO OASTR1.

  P_RESULT = OASTR1.


*获取TOKEN
  CALL FUNCTION 'ZFM_GETOATOKEN'
    IMPORTING
      OUTPUT = OASTR2
    EXCEPTIONS
      OTHERS = 1.
  "zfmdatasave2 'R'.
  CHECK OASTR2 IS NOT INITIAL.
*填入HEADER
  CLEAR:OAMSG,OASTA.
  REFRESH:OAHEAD.
  CLEAR OAHEAD.
  OAHEAD-NAME = 'token'.
  OAHEAD-VALUE = OASTR2.
  APPEND OAHEAD.
*调用函数HTTP
  CLEAR OASTR2.

  PERFORM REPLACE(ZPUBFORM) USING 'https' 'http' CHANGING OAURL.
  CALL FUNCTION 'ZFMS_15_HTTP'
    EXPORTING
      INPUT     = OASTR1
      URL       = OAURL
      REQMETHOD = 'POST' "HTTP 方法
      HTTP1_1   = 'X' "协议1.1/1.0
    IMPORTING
      OUTPUT    = OASTR2 "返回JSON报文
      RTMSG     = OAMSG "消息
      STATUS    = OASTA "HTTP状态
    TABLES
      HEADER    = OAHEAD
    EXCEPTIONS
      OTHERS    = 1.
*返回报文转为SAP结构
  CALL FUNCTION 'ZFMS_14_JSON'
    CHANGING
      JSONSTR = OASTR2
      DATA    = WA_OUTPUT
    EXCEPTIONS
      OTHERS  = 1.

  T_RESULT[] = WA_OUTPUT-DATA-TRESULT.
  P_STATUS = OASTA.
  P_OUTPUT = OAMSG.
  ZFMDATASAVE2 'R'.
ENDFUNCTION.
